public static TDecodableBlock default<TDecodableBlock>(this IDecodableStep<TDecodableBlock> previousDecodableStep)
	where TDecodableBlock : IDecodableStep<TDecodableBlock>
{
	Contract.Requires<ArgumentNullException>(previousDecodableStep != null);

	return
		previousDecodableStep
		.If(decodingContextData => decodingContextData.DecodedValues.FrameLength != 12)
			.RaiseErrorIncorrectFrameLength()
		.EndIf()
		.DecodeBoolean("default", 0.0M).End()
		.DecodeBoolean("default", 0.0M).End()
		.DecodeBytes("default2", 0M, 0M).End()
		.DecodeUnsignedInteger("default2", 0.0M, 0.0M).End()
		.DecodeBytes("default2", 0M, 0M);
}
